// LibAFL support
impl Input for Trace {
    fn generate_name(&self, idx: usize) -> String {
        format!("{id}.trace", id = idx)
    }
}

impl HasLen for Trace {
    fn len(&self) -> usize {
        self.steps.len()
    }
}
